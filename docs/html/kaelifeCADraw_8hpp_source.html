95 void cursorDraw(int strength, int cursorX, int cursorY, int drawRadius, bool drawRandom, CACache::ThreadCache cache) {
96 std::lock_guard<std::mutex> lock(drawBuf.mtx); //make sure drawBuf is not being copied while drawing
97 uint numStates = cache.stateCount;
98
99 if( drawBuf.points.size()!=0 ){//Ensure we aren't checking uninitialized elements
100 //this only discards if this and previous positions are the same.
101 //With big canvas it's unlikely to have two exact same positions,
102 //and calculating the duplicates has marginal performance impact compared to std::find
103 if( ( drawBuf.points.back().pos[0]==cursorX && drawBuf.points.back().pos[1]==cursorY ) ) {
104 return;
105 }
106 }
107
108 uint8_t drawValue=ceil(strength * (numStates - 1) * strength);
109
110 for (int i = -drawRadius; i <= drawRadius; i++) {
111 for (int j = -drawRadius; j <= drawRadius; j++) {
112
113 int mouseDist = i*i+j*j;
114 int radSq = drawRadius*drawRadius;
115 if(mouseDist >= radSq ){continue;} //if outside the circle, discard
116
117 //calculate point of the circle world position
118 uint16_t ax = ((int)cursorX + i + cache.tileRows) % cache.tileRows;
119 uint16_t ay = ((int)cursorY + j + cache.tileCols) % cache.tileCols;
120 if ( alreadyDrawn(ax, ay) ) { continue; }
121 drawSeed++;
122 if(drawRandom) { drawValue = (kaelife::rand(&drawSeed))%numStates; }
123
124 drawnPixel pixelInCircle={
125 .pos={ax,ay},
126 .state = drawValue
127 };

135 .pos={ (uint16_t)cursorX, (uint16_t)cursorY },
