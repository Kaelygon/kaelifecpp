180 uint addPreset(RulePreset preset){
181 preset.name= preset.name=="" ? unsetName : preset.name;
182
183 preset.name.resize(std::min(preset.name.length(), maxNameLength)); //truncate
184 size_t nameLen = preset.name.length();
185
186 uint isDupe = getPresetIndex(preset.name); //check if already exists
187
188 if(isDupe!=UINT_MAX){
189 const size_t ogMaxSfxLen = 4;
190 size_t maxSfxLen = std::min(ogMaxSfxLen,nameLen); //clip suffix to name length
191
192 uint sfxLen = 0; //character length of name suffix number e.g. abc3d123 length is 3, suffix value is 123
193 //if ascii '0' to '9'
194 std::string strNum;
195 strNum.resize(maxSfxLen);
196 do{
197 strNum[sfxLen] = preset.name[nameLen-1-sfxLen];
198 }while( isdigit(strNum[sfxLen]) && (++sfxLen < maxSfxLen) );
199
200 if(sfxLen==0){ //if no index but duplicate exists, add suffix 0
201 preset.name[ std::min(preset.name.length(),maxNameLength) ]='0';
202 }else{ //increment suffix
203 strNum.resize(sfxLen);
204 std::reverse(strNum.begin(),strNum.end());
205 uint sfxValue=stoul(strNum);
206
207 if( ((uint)log10(sfxValue+1)!=(uint)log10(sfxValue)) && sfxValue!=0 ){ //if carry 99 -> 100
208 if(sfxLen==ogMaxSfxLen){ //overflow 9999 -> 0 only if past original max suffix length
209 sfxValue=UINT_MAX;
210 sfxLen=0;
211 nameLen-=maxSfxLen;
212 }
213 sfxLen++; nameLen++; // 999 -> 1000
214 }
215 sfxValue++; //add 1 to the suffix
216
217 preset.name.resize( std::clamp( (size_t)(nameLen-sfxLen), (size_t)0, (size_t)maxNameLength) ); //truncate
218 preset.name.append(std::to_string(sfxValue));
219 }
220 }
221
222// list[list.size()-1].name=preset.name;
223 list.push_back(preset);
224 return list.size()-1;
225 }

388 void randRuleMask( const uint ind, uint maskX=0, uint maskY=0, uint64_t* seed=nullptr ){
389 const uint cellStates=list[ind].stateCount;
390 if(cellStates==0){return;}
391 const uint frac256 = ((uint16_t)UINT8_MAX+cellStates-2)/(cellStates-1);
392 uint64_t* seedPtr = kaelife::rand.validSeedPtr(seed);
393 const uint maxSize=8;
394 maskX = maskX==0 ? (kaelife::rand(seedPtr)%maxSize)+1 : maskX;
395 maskY = maskY==0 ? (kaelife::rand(seedPtr)%maxSize)+1 : maskY;
396
397 list[ind].neigMask.setWidth(maskX);
398 list[ind].neigMask.setHeight(maskY);
399
400 //symmetrize
401 bool symX=kaelife::rand(seedPtr)%4;//75% chance
402 bool symY=kaelife::rand(seedPtr)%4;
403
404 size_t mw=list[ind].neigMask.getWidth();
405 size_t mh=list[ind].neigMask.getHeight();
406 size_t randWidth = (mw+symX)/(symX+1); //Skip elements that will be replaced by symmetry
407 size_t randHeight = (mh+symY)/(symY+1);
408 uint maxDelta = mw*mh;
409
410 const uint multi = 4;
411 const uint divid = 2;
412 for(uint i=0;i<randWidth;++i){
413 for(uint j=0;j<randHeight;++j){
414 uint mwDelta = mw - abs((int)mw/2-(int)i); //distance from mask center
415 uint mhDelta = mh - abs((int)mh/2-(int)j);
416
417 uint maskValue = ceil ( (((float)(kaelife::rand(seedPtr)%cellStates)) / cellStates) * UINT8_MAX ); //randomized mask value
418 maskValue=maskValue*(mwDelta*mhDelta)/(maxDelta)*multi/divid; //scale lower the further from mask center
419 maskValue = (maskValue+frac256-1)/frac256*frac256; //round to nearest cellState multiple scaled to 255
420 maskValue = maskValue > UINT8_MAX ? UINT8_MAX : maskValue;
421 list[ind].neigMask[i][j]=(uint8_t)maskValue;
422 }
423 }
424
425 //symmetrize remainding elements
426 for(uint i=randWidth; i<mw; ++i){
427 if(!symX){break;}
428 for(uint j=0; j<mh; ++j){
429 list[ind].neigMask[i][j]=list[ind].neigMask[mw-i-1][j];
430 }
431 }
432
433 for(uint i=0; i<mw; ++i){
434 if(!symY){break;}
435 for(uint j=randHeight; j<mh; ++j){
436 list[ind].neigMask[i][j]=list[ind].neigMask[i][mh-j-1];
437 }
438 }

450 void randRuleRange(const uint ind, uint16_t minValue, uint16_t maxValue=0, uint64_t* seed=nullptr ) {
451 maxValue = maxValue ? maxValue : calcMaxNeigsum(ind);
452 uint64_t* seedPtr = kaelife::rand.validSeedPtr(seed);
453 uint rangeSize=list[ind].ruleRange.size();
454 list[ind].ruleRange.clear();
455
456 uint current=minValue;
457 uint i;
458 for(i=0; i<rangeSize; ++i){
459 uint16_t addMax = 2*(((maxValue)-current+1)/(rangeSize-i)); //divide maxValue, current delta by remaining elements
460 addMax+=addMax==0; //prevent divide by 0
461 current+= kaelife::rand(seedPtr)%addMax; //add to next range
462 list[ind].ruleRange.push_back(current);
463 if(current>maxValue){current=maxValue; break;} //any element above maxValue isn't used
464 }
