155
156 // Check if already backlogged

168void CABacklog::doBacklog() {
169 std::lock_guard<std::mutex> lock(mtx);
170
171 if (list.empty()) {
172 return;
173 }
174
175 // Flag to prevent cloning buffer every single keyword
176 uint cloneBufferRequest = 0;
177
178 do {
179 const char* keyword = list.back();
180 list.pop_back();
181
182 // Find the waveform function pointers based on the string identifier
183 CAB_func selectedFunction = NULL;
184 for (size_t i = 0; i < keywordMap.size(); ++i) {
185 if (strcmp(keyword, keywordMap[i].id) == 0) {
186 selectedFunction = keywordMap[i].func;
187 break;
188 }
189 }
190 if(selectedFunction==NULL){
191 printf("Invalid backlog key!\n");
192 continue;
193 }
194 cloneBufferRequest = (this->*selectedFunction)();
195
196 }while (!list.empty());
197
198 list.clear();
199
200 if (cloneBufferRequest) {
201 caData.cloneBuffer();
202 }
203}
