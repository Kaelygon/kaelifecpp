119 static void updateTexture(const CAData& cellData) {
120 uint activeRenderBuf = cellData.mainCache.activeBuf; // ensure buffer doesn't change during render
121
122 // allocate memory for pixelData
123 __attribute__((aligned(64))) uint8_t *pixelData = new uint8_t[cellData.mainCache.tileRows * cellData.mainCache.tileCols];
124
125 for (uint i = 0; i < cellData.mainCache.tileRows; ++i) {
126 const auto& row = cellData.cellState[activeRenderBuf].at(i);
127 std::memcpy(pixelData + i * cellData.mainCache.tileCols, row.data(), cellData.mainCache.tileCols);
128 }
129
130 glBindTexture(GL_TEXTURE_2D, textureID);
131
132 glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, cellData.mainCache.tileCols, cellData.mainCache.tileRows, GL_LUMINANCE, GL_UNSIGNED_BYTE, pixelData);
133
134 glBindTexture(GL_TEXTURE_2D, 0);

159 // Function to process shader source and resolve includes
160 std::string processShaderSource(const std::string& source) {
161 std::string processedSource = source;
162
163 // Define a special marker for includes
164 const std::string includeMarker = "//#include";
165
166 // Find and process includes
167 size_t pos = processedSource.find(includeMarker);
168 while (pos != std::string::npos) {
169 // Find the end of the line
170 size_t endOfLine = processedSource.find("\n", pos);
171 if (endOfLine != std::string::npos) {
172 // Extract the file path from the include directive
173 std::string includeDirective = processedSource.substr(pos, endOfLine - pos);
174 size_t filePathStart = includeDirective.find("\"") + 1;
175 size_t filePathEnd = includeDirective.rfind("\"");
176 std::string filePath = includeDirective.substr(filePathStart, filePathEnd - filePathStart);
177
178 // Read the content of the included file
179 std::string includedContent = readShaderFile(filePath);
180
181 // Replace the include directive with the file content
182 replaceAll(processedSource, includeDirective, includedContent);
183 }
184
185 // Find the next include
186 pos = processedSource.find(includeMarker, pos + 1);
187 }
188
189 return processedSource;
190 }

192 // Function to compile a shader
193 GLuint compileShader(GLenum type, const std::string& source) {
194 GLuint shader = glCreateShader(type);
195
196 // Process shader source to resolve includes
197 std::string processedSource = processShaderSource(source);
198 const char* sourceCStr = processedSource.c_str();
199
200 glShaderSource(shader, 1, &sourceCStr, nullptr);
201 glCompileShader(shader);
202
203 // Check compilation status

206 if (!success) {

209 std::cerr << "Shader compilation error:\n" << infoLog << std::endl;
210 glDeleteShader(shader);

217 // Function to create a shader program and link shaders
218 GLuint createShaderProgram(const std::string& vertexShaderSource, const std::string& fragmentShaderSource) {
219 GLuint vertexShader = compileShader(GL_VERTEX_SHADER, vertexShaderSource);
220 GLuint fragmentShader = compileShader(GL_FRAGMENT_SHADER, fragmentShaderSource);
221
222 // Create shader program
223 GLuint shaderProgram = glCreateProgram();
224 glAttachShader(shaderProgram, vertexShader);
225 glAttachShader(shaderProgram, fragmentShader);
226 glLinkProgram(shaderProgram);
227
228 GLint success;
229 glGetProgramiv(shaderProgram, GL_LINK_STATUS, &success);
230 if (!success) {
231 GLchar infoLog[512];
232 glGetProgramInfoLog(shaderProgram, sizeof(infoLog), nullptr, infoLog);
233 std::cerr << "Shader program linking error: " << infoLog << std::endl;
234 glDeleteProgram(shaderProgram);
235 return 0;
236 }
237
238 // Delete individual shaders, as they are now part of the program
239 glDeleteShader(vertexShader);
240 glDeleteShader(fragmentShader);
241
242 return shaderProgram;
243 }
