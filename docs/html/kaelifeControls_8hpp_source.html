364void InputHandler::detectInput() {
365 holdAccel=1.0;
366 while(!QUIT_FLAG){
367 SDL_Event event;
368
369 timerSt=SDL_GetTicks();
370
371 while(SDL_PollEvent(&event) != 0){
372
373 //wrap and track mouse
374 SDL_GetMouseState(&cursorPos[0], &cursorPos[1]);
375
376
377
378 if (event.type == SDL_QUIT) {
379 QUIT_FLAG = true;
380 } else if (event.type == SDL_KEYDOWN || event.type == SDL_KEYUP) {
381 if((event.type == SDL_KEYDOWN)){//accelerator increment
382 holdAccel+=0.25;
383 }else{
384 holdAccel=1.0;
385 }
386
387 // Store the state of the key
388 keyStates[event.key.keysym.sym] = (event.type == SDL_KEYDOWN);
389
390
391 SDL_Keymod modState = SDL_GetModState();
392 SDL_Keycode combinedKey = event.key.keysym.sym | (modState<<16);
393
394 if(event.type == SDL_KEYDOWN){
395 if(kaelife::INPUT_DEBUG){
396 printf("Key: %d, Modifiers: %d, Result: %d \n", event.key.keysym.sym, (modState<<16), combinedKey);
397 }
398 // Find the associated function in the map
399 auto it = keyFuncMap.find(combinedKey);
400
401 // If the key combination is found, execute the associated function
402 if (it != keyFuncMap.end()) {
403 std::function<void(CAData&)>& func = it->second;
404 func(cellData);
405 }
406 }
407 }
408
409 if (event.type == SDL_MOUSEBUTTONDOWN || event.type == SDL_MOUSEBUTTONUP) {
410 // Store the state of the mouse button
411 keyStates[event.button.button] = (event.type == SDL_MOUSEBUTTONDOWN);
412 windowFocus=keyStates[event.button.button];
413
414 if (kaelife::INPUT_DEBUG) {
415 printf("Mouse button %d pressed\n", event.button.button);
416 }
417 if(kaelife::INPUT_DEBUG){
418 printf("Mouse pos (%d, %d)\n", cursorPos[0], cursorPos[1]);
419 }
420 if(kaelife::INPUT_DEBUG){
421 auto cursorPosW = getWorldCursorPos();
422 printf("Mouse world (%d, %d)\n", cursorPosW[0], cursorPosW[1]);
423 }
424 }
425 }
426
427 //run these every call
428 if (keyStates[SDL_BUTTON_LEFT] || keyStates[SDL_BUTTON_RIGHT] || event.type == SDL_MOUSEBUTTONDOWN) {
429 auto cursorPos = getWorldCursorPos();
430 if(kaelife::INPUT_DEBUG){
431 printf("cursorPos[0] %d\n",cursorPos[0]);
432 printf("cursorPos[1] %d\n",cursorPos[1]);
433 }
434 // Mouse motion or left button pressed
435 int strength = keyStates[SDL_BUTTON_LEFT] ? 1 : 0;
436 cellData.kaeDraw.cursorDraw(strength, cursorPos[0], cursorPos[1], drawRadius, drawRandom, cellData.mainCache);
437 cellData.backlog->add("cursorDraw");
438 }
439
440 if (windowFocus){
441 int windowWrapLimit[2];
442 SDL_GetWindowSize(SDLWindow, &windowWrapLimit[0], &windowWrapLimit[1]);
443 windowWrapLimit[0]-=1;
444 windowWrapLimit[1]-=1;
445
446 bool moveCursor=0;
447 //if past window edges
448 if ( cursorPos[0]>= windowWrapLimit[0] ) {cursorPos[0]=0; moveCursor=true;}
449 else
450 if ( cursorPos[0]<= 0 ) {cursorPos[0]=windowWrapLimit[0]; moveCursor=true;}
451
452 if ( cursorPos[1]>= windowWrapLimit[1] ) {cursorPos[1]=0; moveCursor=true;}
453 else
454 if ( cursorPos[1]<= 0 ) {cursorPos[1]=windowWrapLimit[1]; moveCursor=true;}
455
456 if(moveCursor){
457 SDL_WarpMouseInWindow(SDLWindow, cursorPos[0], cursorPos[1]);
458 }
459 }
460
461 do{ //update twice a frame
462 timerNd=SDL_GetTicks() - timerSt;
463 SDL_Delay(1);
464 }while(timerNd < cellData.targetFrameTime/2);
465 }
466}

472 std::array<double, 4> InputHandler::getWorldTransform() {
473 int x, y;
474 SDL_GetWindowSize(SDLWindow, &x, &y );
475
476 double windowAspectRatio = (double)(x) / (double)(y);
477
478 double offsetX = 0;
479 double offsetY = 0;
480 double scaledX = x;
481 double scaledY = y;
482
483 //Height or width restricted
484 if (windowAspectRatio > cellData.aspectRatio) {
485 // Calculate the scale and offset for black bars
486 scaledX = y * cellData.aspectRatio;
487 offsetX = (x - scaledX) / 2.0;
488 } else {
489 scaledY = x / cellData.aspectRatio;
490 offsetY = (y - scaledY) / 2.0;
491 }
492
493 std::array<double, 4> output = {
494 offsetX,
495 offsetY,
496 scaledX,
497 scaledY
498 };
499
500 return output;
501
502 }

504 //mouse world coordinates wrapped
505 std::array<int, 2> InputHandler::getWorldCursorPos() {
506 auto offsetScale = getWorldTransform();
507 std::vector<float> worldCursorPos(2);
508
509 //black bar offset
510 worldCursorPos[0]=cursorPos[0]-offsetScale[0];
511 worldCursorPos[1]=cursorPos[1]-offsetScale[1];
512
513 //normalize to 0-1.0
514 worldCursorPos[0] = (((float)worldCursorPos[0]) / (float)(offsetScale[2]));
515 worldCursorPos[1] = 1.0 - (((float)worldCursorPos[1]) / (float)(offsetScale[3])); //glew vs sdl have inverted Y
516
517 //scale to tile grid
518 worldCursorPos[0] *= cellData.mainCache.tileRows;
519 worldCursorPos[1] *= cellData.mainCache.tileCols;
520
521 std::array<int, 2> worldCursorOut;
522 worldCursorOut[0]=(int)(worldCursorPos[0]+cellData.mainCache.tileRows)%cellData.mainCache.tileRows;
523 worldCursorOut[1]=(int)(worldCursorPos[1]+cellData.mainCache.tileCols)%cellData.mainCache.tileCols;
524 return worldCursorOut;
